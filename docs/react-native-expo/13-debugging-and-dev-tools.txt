                    4. CONSOLE LOGGING
================================================================================

The simplest debugging tool - use console methods effectively.

CONSOLE METHODS:
----------------

    // Basic logging
    console.log('Simple message');
    console.log('User:', user);
    console.log('Items:', items, 'Count:', items.length);
    
    // Warnings (yellow)
    console.warn('This might be a problem');
    
    // Errors (red)
    console.error('Something went wrong:', error);
    
    // Information
    console.info('FYI:', info);
    
    // Grouped output
    console.group('User Details');
    console.log('Name:', user.name);
    console.log('Email:', user.email);
    console.groupEnd();
    
    // Table format (great for arrays/objects)
    console.table(users);
    console.table(user, ['name', 'email']); // Specific columns
    
    // Timing
    console.time('fetchUsers');
    await fetchUsers();
    console.timeEnd('fetchUsers'); // "fetchUsers: 234ms"
    
    // Count occurrences
    function render() {
      console.count('render'); // "render: 1", "render: 2", etc.
    }
    
    // Stack trace
    console.trace('How did we get here?');
    
    // Clear console
    console.clear();


EFFECTIVE LOGGING TIPS:
-----------------------

    // 1. Use labels for context
    console.log('[AuthService] User logged in:', user.id);
    console.log('[HomeScreen] Rendering with items:', items.length);
    
    // 2. Use emoji for visibility 👀
    console.log('🔥 Hot reload triggered');
    console.log('✅ User authenticated');
    console.log('❌ Error:', error);
    console.log('⏱️ API call took:', duration);
    
    // 3. Log objects properly
    // ❌ Doesn't show object details
    console.log('User: ' + user);
    
    // ✅ Shows full object
    console.log('User:', user);
    console.log('User:', JSON.stringify(user, null, 2));
    
    // 4. Conditional logging
    const DEBUG = __DEV__;
    DEBUG && console.log('Debug info:', data);


REMOVING LOGS IN PRODUCTION:
----------------------------

    // babel.config.js
    module.exports = function (api) {
      api.cache(true);
      return {
        presets: ['babel-preset-expo'],
        plugins: [
          'react-native-reanimated/plugin',
          // Remove console.* in production
          process.env.NODE_ENV === 'production' && 
            ['transform-remove-console', { exclude: ['error', 'warn'] }],
        ].filter(Boolean),
      };
    };

Install: `npm install babel-plugin-transform-remove-console --save-dev`

USING BREAKPOINTS:
------------------

    function handleLogin(email: string, password: string) {
      // Click the line number in VS Code/Chrome to set breakpoint
      debugger; // Or use debugger statement
      
      const user = validateCredentials(email, password);
      // Execution will pause here if breakpoint is set
      
      if (!user) {
        throw new Error('Invalid credentials');
      }
      
      return user;
    }


DEBUGGING ASYNC CODE:
---------------------

    async function fetchUserData() {
      try {
        console.log('1. Starting fetch...');
        debugger; // Pause here
        
        const response = await fetch('/api/user');
        console.log('2. Got response:', response.status);
        
        const data = await response.json();
        console.log('3. Parsed data:', data);
        
        return data;
      } catch (error) {
        console.log('ERROR:', error);
        debugger; // Pause on error
        throw error;
      }
    }

                    8. NETWORK DEBUGGING

Debug API calls and network requests.

LOGGING NETWORK REQUESTS:
-------------------------

    // Enable network request logging
    // Add to your entry file (e.g., index.js or App.tsx)
    if (__DEV__) {
      // XMLHttpRequest logging
      const originalXHROpen = XMLHttpRequest.prototype.open;
      XMLHttpRequest.prototype.open = function(...args) {
        console.log('[XHR]', args[0], args[1]);
        return originalXHROpen.apply(this, args);
      };
    }


USING AXIOS INTERCEPTORS:
-------------------------

    // Debug all axios requests
    import axios from 'axios';
    
    if (__DEV__) {
      axios.interceptors.request.use((config) => {
        console.log('🌐 Request:', config.method?.toUpperCase(), config.url);
        console.log('   Body:', config.data);
        return config;
      });
      
      axios.interceptors.response.use(
        (response) => {
          console.log('✅ Response:', response.status, response.config.url);
          return response;
        },
        (error) => {
          console.log('❌ Error:', error.response?.status, error.config?.url);
          console.log('   Message:', error.message);
          return Promise.reject(error);
        }
      );
    }

OPTIMIZING LISTS:
-----------------

    // ❌ Bad - renders all items
    <ScrollView>
      {items.map(item => <Item key={item.id} data={item} />)}
    </ScrollView>
    
    // ✅ Good - virtualizes, only renders visible items
    <FlatList
      data={items}
      renderItem={({ item }) => <Item data={item} />}
      keyExtractor={item => item.id}
      getItemLayout={(data, index) => ({
        length: ITEM_HEIGHT,
        offset: ITEM_HEIGHT * index,
        index,
      })}
      maxToRenderPerBatch={10}
      windowSize={5}
      removeClippedSubviews={true}
    />


MEMOIZATION:
------------

    import { memo, useMemo, useCallback } from 'react';
    
    // Memoize component to prevent re-renders
    const ExpensiveItem = memo(function ExpensiveItem({ data, onPress }) {
      return (
        <Pressable onPress={onPress}>
          <Text>{data.title}</Text>
        </Pressable>
      );
    });
    
    function ParentComponent({ items }) {
      // Memoize computed values
      const sortedItems = useMemo(() => {
        return [...items].sort((a, b) => a.name.localeCompare(b.name));
      }, [items]);
      
      // Memoize callbacks
      const handlePress = useCallback((id) => {
        console.log('Pressed:', id);
      }, []);
      
      return (
        <FlatList
          data={sortedItems}
          renderItem={({ item }) => (
            <ExpensiveItem 
              data={item} 
              onPress={() => handlePress(item.id)} 
            />
          )}
        />
      );
    }

RED SCREEN ERRORS:
------------------

    ┌─────────────────────────────────────────────────────────────────────────┐
    │                    COMMON RED SCREEN ERRORS                              │
    ├─────────────────────────────────────────────────────────────────────────┤
    │                                                                          │
    │   "Text strings must be rendered within a <Text> component"             │
    │   └── You have text outside <Text> tags                                 │
    │       ❌ <View>Hello</View>                                             │
    │       ✅ <View><Text>Hello</Text></View>                                │
    │                                                                          │
    │   "Invariant Violation: Invalid hook call"                              │
    │   └── Hook used outside component or conditionally                      │
    │       ❌ if (x) { const [a, b] = useState() }                           │
    │       ✅ const [a, b] = useState()                                      │
    │                                                                          │
    │   "Cannot read property 'X' of undefined"                               │
    │   └── Accessing property of undefined variable                          │
    │       ❌ user.name (when user is undefined)                             │
    │       ✅ user?.name or user && user.name                                │
    │                                                                          │
    │   "Maximum call stack size exceeded"                                    │
    │   └── Infinite loop or recursion                                        │
    │       Check useEffect dependencies, recursive calls                     │
    │                                                                          │
    │   "Network request failed"                                              │
    │   └── Can't reach API (check URL, network, CORS)                        │
    │       iOS: Allow HTTP if not HTTPS (App Transport Security)             │
    │                                                                          │
    └─────────────────────────────────────────────────────────────────────────┘

    npx expo install --fix
    
    # Check for outdated packages
    npx expo install --check


SIMULATOR/EMULATOR ISSUES:
--------------------------

    iOS Simulator:
    - Reset: Device → Erase All Content and Settings
    - Clean build: Cmd+Shift+K in Xcode
    
    Android Emulator:
    - Cold boot: AVD Manager → Cold Boot Now
    - Wipe data: AVD Manager → Wipe Data
    - Check ADB: adb devices


HOT RELOAD NOT WORKING:
-----------------------

    1. Check file is saved
    2. Check for syntax errors
    3. Check Fast Refresh is enabled (Dev Menu)
    4. Restart Metro: npx expo start --clear
    5. Reinstall app on device
